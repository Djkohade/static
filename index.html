function loadPage(page) { if (!page) { document.getElementById('content').textContent = 'Error #1'; return; } if (!/^[a-zA-Z0-9-_]+$/.test(page)) { document.getElementById('content').textContent = 'Error #2'; return; } document.getElementById('loader').style.display = 'inline'; document.getElementById('content').textContent = ''; setTimeout(() => { fetch(`pages/${page}.html`) .then(response => { if (!response.ok) throw new Error('404'); return response.text(); }) .then(html => { document.getElementById('loader').style.display = 'none'; document.getElementById('content').innerHTML = html; }) .catch(err => { document.getElementById('loader').style.display = 'none'; //document.getElementById('content').textContent = 'Error: ' + err.message; }); }, 3000); // задержка 3000 мс = 3 секунды }
